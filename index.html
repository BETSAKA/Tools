Comparing Protected Areas selection in 3 BETSAKA Working Units

Code used for PA selection

The code used to select the protected areas (PAs) analysed in 3 working packages (WU1.1 on deforestation, WU1.2 on fires, and WU2.1 on livelihoods) is in: - wu_snippets\deforestation_01-treatment-assignation.qmd - wu_snippets\fire_01_data_acquisition.qmd - wu_snippets\livelihoods_02_treatment_status.qmd

Code in WU 1.1

  • Source data: July 2025 WDPA extract for Madagascar and data/legal_texts.rds.
  • PA selection: designated PAs plus PAs identified through the legal review; polygon and multipolygon geometries intersecting Madagascar.
  • Modifications: replace STATUS_YR with the earliest legal year when available and set Ambohimirahavavy Marivorahona (WDPAID == 555697871) to 2015.
Code 
library(tidyverse, quietly = TRUE)
library(sf, quietly = TRUE)
library(wdpar, quietly = TRUE)
library(geodata, quietly = TRUE)
library(gt, quietly = TRUE)

# Systèmes de coordonnées de référence 
standard_crs <- 4326
mdg_crs <- 29702 

# Load the work implemented in: https://betsaka.github.io/conservation-dynamic-madagascar/03_legal_docs.html
legal_texts <- read_rds("data/legal_texts.rds")

# Use the legal analysis made from CNLEGIS to identify early protections
# that are not registered in WDPA: 
legal_years <- legal_texts |>
  filter(str_detect(type_decision, "creation_definitive|mise_protection_temporaire")) |>
  mutate(
    date_texte = as.Date(date_texte),
    legal_year = year(date_texte)
  ) |>
  group_by(WDPAID) |>
  summarize(
    legal_year = min(legal_year, na.rm = TRUE),
    .groups = "drop"
  )


# Load WDPA data
wdpa <- wdpa_read("data/WDPA_WDOECM_Jul2025_Public_MDG.zip") %>%
 # wdpa_clean() |> # This might cause to drop existing PAs!
 # Include PAs that exist according to the legal analysis
  filter(STATUS == "Designated" | WDPAID %in% legal_years$WDPAID) %>%
  mutate(MARINE = recode(MARINE,
                         "0" = "Terrestre",
                         "1" = "Mixte",
                         "2" = "Marine"),
        GEOMETRY_TYPE = st_geometry_type(.)) %>%
  filter(GEOMETRY_TYPE %in% c("POLYGON", "MULTIPOLYGON"),)

# Make sure join keys have compatible types
legal_years <- legal_years |> mutate(WDPAID = as.numeric(WDPAID))
wdpa <- wdpa |> mutate(WDPAID = as.numeric(WDPAID))

# Join and replace STATUS_YR where a legal year exists
wdpa <- wdpa |>
  left_join(legal_years, by = "WDPAID") |>
  mutate(
    STATUS_YR = if_else(!is.na(legal_year), as.integer(legal_year), STATUS_YR)
  ) |>
  select(-legal_year)

# Load boundary
contour_mada <- gadm(country = "Madagascar", level = 0, path = "data") %>%
  st_as_sf() %>%
  st_set_crs(standard_crs)

# Intersection des AP de WDPA avec la limite de Madagascar
wdpa_terrestre <- wdpa %>% 
  st_transform(crs = standard_crs) %>%
  st_make_valid() %>%
  st_intersection(contour_mada) %>% 
  st_transform(mdg_crs) %>%
  mutate(
    area_m2  = as.numeric(st_area(.)),
    area_ha  = area_m2 / 1e4,
    area_km2 = area_m2 / 1e6
  ) %>%
  st_transform(standard_crs)

# Ajout de la date de création du complexe des aires protégées Ambohimirahavavy Marivorahona 

pa_wu11 <- wdpa_terrestre %>%
  mutate(STATUS_YR = ifelse(WDPAID == 555697871, 2015, STATUS_YR))

# clean intermediary objects
rm(codelist, contour_mada, legal_texts, legal_years, wdpa, wdpa_terrestre)

Code in WU 1.2

  • Source data: data/dynamic_wdpa.gpkg.
  • PA selection: terrestrial and mixed PAs (MARINE %in% 0:1) with Ramsar and inscribed records removed.
  • Modifications: derive STATUS_YR_DYN, collapse multiple status rows to one row per PA, set Ambohimirahavavy Marivorahona (WDPAID == 555697871) to 2015, and harmonize names, IUCN categories, and management authority fields.

Code in WU 2.1

  • Source data: October 2025 WDPA extract and corrected SAPM 2017 shapefile.
  • PA selection: designated PAs after excluding UNESCO-MAB Biosphere Reserves, Ramsar sites, and World Heritage sites.
  • Modifications: standardize selected PA names, set Ambohimirahavavy Marivorahona (WDPAID == 555697871) to 2015, and replace WDPA creation years with SAPM 2017 years when the two sources disagree.

  filter(STATUS == "Designated") %>%

                         "terrestrial" = "Terrestre",

                         "partial" = "Mixte")) %>%
  filter(DESIG != "UNESCO-MAB Biosphere Reserve") %>%
  filter(DESIG != "Ramsar Site, Wetland of International Importance") %>%
  filter(DESIG != "World Heritage Site (natural or mixed)") %>%

  mutate(

    WDPA_YEAR = STATUS_YR
  ) %>%

normalize_pa_name <- function(x) {
  x |>
    stringr::str_to_upper() |>
    stringr::str_replace_all("[-_']", " ") |>
    stringr::str_replace_all("[^A-Z0-9 ]", " ") |>
    stringr::str_squish()
}

prep_pa_register <- function(data, wu, year_col = STATUS_YR) {
  data |>
    sf::st_drop_geometry() |>
    transmute(
      wu = wu,
      WDPAID = as.character(WDPAID),
      NAME = as.character(NAME),
      NAME_STD = normalize_pa_name(NAME),
      STATUS_YR = suppressWarnings(as.integer({{ year_col }}))
    ) |>
    distinct(WDPAID, .keep_all = TRUE)
}
